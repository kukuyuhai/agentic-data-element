### 交付物 2：SQL 分析——爆款品类识别
```sql
-- 近 90 天 GMV Top 10 品类环比增速
WITH cur AS (
  SELECT category, SUM(gmv) AS gmv
  FROM trades
  WHERE ds BETWEEN CURRENT_DATE - INTERVAL '90 day' AND CURRENT_DATE
  GROUP BY category
),
prev AS (
  SELECT category, SUM(gmv) AS gmv
  FROM trades
  WHERE ds BETWEEN CURRENT_DATE - INTERVAL '180 day' AND CURRENT_DATE - INTERVAL '91 day'
  GROUP BY category
)
SELECT c.category,
       c.gmv AS cur_gmv,
       p.gmv AS prev_gmv,
       ROUND((c.gmv - p.gmv) * 100.0 / NULLIF(p.gmv, 0), 2) AS growth_pct
FROM cur c
LEFT JOIN prev p USING (category)
ORDER BY cur_gmv DESC
LIMIT 10;
```

### 交付物 3：Python 时间序列——GMV 预测
```python
import pandas as pd
from statsmodels.tsa.holtwinters import ExponentialSmoothing

df = pd.read_csv("daily_gmv.csv", parse_dates=["ds"])
ts = df.set_index("ds")["gmv"].asfreq("D").fillna(method="ffill")

model = ExponentialSmoothing(ts, trend="add", seasonal="add",
                             seasonal_periods=7).fit()
forecast = model.forecast(30)          # 未来 30 天
print(forecast.round(2))
```
